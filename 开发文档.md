## 一个折线|柱状图

输入：

```js
dataSource: [
    {
        label: "x1",
        value: 100,
    },
    {
        label: "x2",
        value: 120,
    },
    {
        label: "x3",
        value: 110,
    },
    {
        label: "x4",
        value: 150,
    },
];
```

输出：label 作为 X 轴

```js
    xAxis: {
        type: 'category',
        data: ['X1', 'X2', 'X3', 'X4']
    },
    yAxis: {
        type: 'value'
    },
    series: [{
        data: [150, 230, 224, 218, 135, 147, 260],
        type: 'line'
    }]
```

## 两条折线|柱状图

输入：

```js
dataSource = [
    {
        series: "学生",
        label: "x1",
        value: 100,
    },
    {
        series: "学生",
        label: "x2",
        value: 120,
    },
    {
        series: "学生",
        label: "x3",
        value: 110,
    },
    {
        series: "教师",
        label: "x1",
        value: 150,
    },
    {
        series: "教师",
        label: "x2",
        value: 230,
    },
    {
        series: "教师",
        label: "x3",
        value: 224,
    },
];
```

输出：

```js
    xAxis: {
        type: 'category',
        data: ['X1', 'X2', 'X3']
    },
    yAxis: {
        type: 'value'
    },
    series: [
        {
            name: '学生',
            data: [100, 120, 110],
            type: 'line'
        },
        {
            name: '教师',
            data: [150, 230, 224],
            type: 'line'
        },
    ]
```

## 雷达图 (单个数据)

输入：

```js
dataSource = [
    {
        label: "指标1",
        value: 10,
    },
    {
        label: "指标2",
        value: 12,
    },
    {
        label: "指标3",
        value: 8,
    },
    {
        label: "指标4",
        value: 13,
    },
    {
        label: "指标5",
        value: 9,
    },
];
```

输出：

```js
data: [
    {
        value: [10, 12, 8, 13, 9],
    },
];
```

## 雷达图(多系列数据)

输入：

```js
dataSource = [
    {
        label: "指标1",
        value: 10,
        series: "小汽车排放",
    },
    {
        label: "指标2",
        value: 12,
        series: "小汽车排放",
    },
    {
        label: "指标3",
        value: 8,
        series: "小汽车排放",
    },
    {
        label: "指标4",
        value: 13,
        series: "小汽车排放",
    },
    {
        label: "指标1",
        value: 9,
        series: "飞机排放",
    },
    {
        label: "指标2",
        value: 23,
        series: "飞机排放",
    },
    {
        label: "指标3",
        value: 13,
        series: "飞机排放",
    },
    {
        label: "指标4",
        value: 20,
        series: "飞机排放",
    },
];
```

输出
```js
series:[
    {
        name: "小汽车排放",
        type:"radar",
        data:[10, 12, 8, 4]
    },
    {
        name: "飞机排放",
        type:"radar",
        data:[9, 13, 23, 20]
    },
]
```
